1 describe("Ribs.backbone.support.modelSupport", function () {
  2     var models, model, mixin;
  3 
  4     beforeEach(function () {
  5         models = new Backbone.Model();
  6         model = new Backbone.Model({
  7             test: "foo",
  8             myAttr: "1234"
  9         });
 10         mixin = _.extend({}, Ribs.backbone.mixins.support.model.modelBase);
 11         mixin.pivot = mixin;
 12         mixin.models = models;
 13         mixin.backboneModels = { modelChooser: model };
 14         mixin.mixinInitialize();
 15     });
 16 
 17     it("Should create model internal automatically", function () {
 18         expect(mixin.getModelJSON({ jsonModelName: "internal" })).toBeDefined();
 19     });
 20 
 21     it("Should have the given model", function () {
 22         expect(mixin.getModelJSON({ jsonModelName: "modelChooser" })).toBeDefined();
 23     });
 24 
 25     describe("Changing models", function () {
 26         var callStack;
 27 
 28         beforeEach(function () {
 29             callStack = objectCallObserver(mixin);
 30             callStack.start();
 31         });
 32 
 33         afterEach(function () {
 34             callStack.expectFinished();
 35         });
 36 
 37         it("Should inform about model replaced by the user", function () {
 38             callStack.expectCalls([
 39                 "modelChangeHandler",
 40                 "modelRemoved",
 41                 "modelAdded"
 42             ]);
 43             models.set({ modelChooser: new Backbone.Model() });
 44         });
 45 
 46         it("Should inform about model added by the user", function () {
 47             callStack.expectCalls([
 48                 "modelChangeHandler",
 49                 "modelAdded"
 50             ]);
 51             models.set({ newModel: new Backbone.Model() });
 52         });
 53 
 54         it("Should inform about model removed by the user", function () {
 55             callStack.expectCalls([
 56                 "modelChangeHandler",
 57                 "modelRemoved"
 58             ]);
 59             models.unset("modelChooser");
 60         });
 61     });
 62 
 63     it("Should mixins multiple models", function () {
 64         models.set({ newModel: new Backbone.Model({ val: "text", test: "test" }) });
 65         mixin.jsonModelName = ["newModel", "modelChooser"];
 66         Ribs.mixins.support.functions.resolveJSON.apply(mixin);
 67         expect(mixin.json.val).toEqual("text");
 68         expect(mixin.json.test).toEqual("foo");
 69         expect(mixin.json.myAttr).toEqual("1234");
 70     });
 71 });
